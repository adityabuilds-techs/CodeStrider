Student Result Calculator

📄 Project Code

name = input("Enter Name: ")
age = int(input("Enter Age: "))
marks = int(input("Enter Marks: "))

def result(marks):
    if marks >= 35:
        return "Pass"
    else:
        return "Fail"

print("Name:", name)
print("Age:", age)
print("Marks:", marks)
print("Result:", result(marks))
